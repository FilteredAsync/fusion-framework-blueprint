local Fusion = require(ReplicatedStorage.Fusion)
local scoped, peek, doCleanup = Fusion.scoped, Fusion.peek, Fusion.doCleanup

local scope = scoped(Fusion)
local health = scope:Value(100)

table.insert(scope, function() print("cleanup A") end)
table.insert(scope, function() print("cleanup B") end)

local inner = scope:innerScope()
table.insert(inner, function() print("cleanup inner") end)

local detached = scope.deriveScope(scope)
table.insert(detached, function() print("cleanup detached") end)

doCleanup(scope)
-- > cleanup inner    (inner scopes die with their parent)
-- > cleanup B        (then the parent, reverse creation order)
-- > cleanup A
-- "cleanup detached" did NOT print: deriveScope shares methods, not lifetime

health:set(50) -- silent no-op: destroyed objects never error, they just stop working

doCleanup(detached) -- > cleanup detached

local coins = scope:Value(100)
local price = scope:Value(75)

local evaluations = 0
local canAfford = scope:Computed(function(use)
    evaluations += 1
    return use(coins) >= use(price)
end)

print(evaluations)     -- > 0     lazy: nothing has run yet
print(peek(canAfford)) -- > true
print(evaluations)     -- > 1     the first read evaluated it

coins:set(100)         -- same value: the similarity check skips the update
peek(canAfford)
print(evaluations)     -- > 1     no recompute

coins:set(10)
print(peek(canAfford)) -- > false
print(evaluations)     -- > 2

-- peek() inside a computed does NOT register a dependency:
local label = scope:Computed(function(use)
    return `{use(coins)} / {peek(price)} coins`
end)
peek(label)
price:set(50)          -- label never re-evaluates for price changes (verified)

-- Observers are the side-effect primitive:
local obs = scope:Observer(coins)
obs:onChange(function()
    print("coins changed to", peek(coins))
end)
obs:onBind(function()   -- fires immediately, then on every change
    print("bound at", peek(coins))
end)
-- > bound at 10

coins:set(25)
-- > bound at 25
-- > coins changed to 25
-- all of these connections disconnect automatically at doCleanup(scope)

-- For-objects: incremental collection transforms
local items = scope:Value(table.freeze({ "sword", "shield" }))

local labels = scope:ForValues(items, function(use, itemScope, name)
    print("processing", name)
    table.insert(itemScope, function() print("cleaned", name) end)
    return name:upper()
end)
peek(labels)
-- > processing sword / processing shield (order not guaranteed)

items:set(table.freeze({ "sword", "shield", "bow" }))
peek(labels)
-- > processing bow   (ONLY the new value runs the processor)

items:set(table.freeze({ "sword", "bow" }))
peek(labels)
-- "cleaned shield" runs. Verified caveat: sibling scopes can also be
-- recycled here without reprocessing, so treat per-item cleanup timing
-- as an implementation detail, not a removal signal (§8).

local settings = scope:Value(table.freeze({ music = 0.8, sfx = 0.5 }))
local byLabel = scope:ForPairs(settings, function(use, s, key, volume)
    return key:upper(), math.round(volume * 100)
end)
print(peek(byLabel).MUSIC) -- > 80

local box = scope:New "TextBox" {
    Name = "Greeting",
    Text = text,                      -- state binds; re-assigns on every :set()
    Size = UDim2.fromOffset(200, 40),

    [OnEvent "Focused"] = function()
        print("focused")
    end,
    [OnChange "Text"] = function(new)
        print("text is now", new)     -- fires with the new value
    end,
    [Out "Text"] = typed,             -- player edits flow back into `typed`
    [Attribute "Validated"] = scope:Computed(function(use)
        return #use(typed) > 0        -- attributes bind reactively too
    end),

    [Children] = {
        scope:New "UICorner" {},
    },
}

text:set("Hi")
print(box.Text)  -- > Hi    on this same line: bindings apply synchronously

box.Text = "typed!"
-- next frame: peek(typed) == "typed!" and box:GetAttribute("Validated") == true (verified)

-- Hydrate binds onto an existing instance WITHOUT taking ownership:
local sign = scope:Hydrate(existingLabel) {
    Text = text,        -- sign.Text is now "Hi" and tracks `text`
}

doCleanup(scope)
-- box (New) is destroyed; existingLabel (Hydrate) survives, bindings removed (verified)

1 · Version reality check

verified 3–0The current release of Fusion is v0.3-beta, published 2024-08-30: GitHub's designated "Latest" and the only non-prerelease tag. A 0.4 exists only as an unreleased dev line (0.4.0-dev1 in wally.toml) with draft docs. Everything in this report targets 0.3.

Sources: dphfox/Fusion releases, repo

communityMaintenance status, from the author directly (Roblox OSS Discord, June 20, 2026): "No features are being worked on, but I vaguely keep an eye on things in here to make sure things still tick along," and, having joined Roblox, "my attention has been elsewhere in the ecosystem." Plan accordingly: treat v0.3-beta as the long-term target, expect fixes to come from the community rather than upstream, and do not gate your framework on 0.4 shipping.

2 · Scopes: the core mental model

verified 3–0Fusion 0.3's foundational primitive is the scope: a plain Luau array that groups objects for collective destruction. The author describes it as a generalization of the Maid pattern, with automatic "time travelling" lifetime analysis (the possiblyOutlives warning fires when a value could outlive something it depends on). Every stateful constructor (Value, Computed, Observer, For-objects, Tween, Spring, New, Hydrate) takes a scope as its first argument. The sole exception is Contextual, which is non-destructible.

Sources: scopes tutorial, 0.3 release notes, discussion #292

-- The idiomatic setup. scoped() makes constructors callable as methods.
local Fusion = require(ReplicatedStorage.Packages.Fusion)
local scoped, peek, doCleanup = Fusion.scoped, Fusion.peek, Fusion.doCleanup

local scope = scoped(Fusion)
local health = scope:Value(100)
local label = scope:Computed(function(use)
    return `HP: {use(health)}`
end)

doCleanup(scope) -- everything above is destroyed, reverse order

verified 3–0Three scope constructors with distinct lifetime semantics:

  • scoped(Fusion, ...): new root scope; passed tables become method tables. The docs explicitly recommend this syntax.
  • scope:innerScope(): child scope, destroyed automatically with its parent but cleanable early. Documented as "often the safest choice for creating new scopes." This is your screen/feature lifetime tool.
  • scope.deriveScope: copies the method table only; the result has a completely independent lifecycle. Use when you need the same conveniences without tying lifetimes.

verified 3–0doCleanup disposes of contents in reverse creation order and dispatches on type: functions are called, objects with :destroy()/:Destroy() have it called, Instances are destroyed, connections disconnected, nested tables recursed. The hard constraint: destroyed content is irrecoverable. The docs say to remove all references and never use them again.

Lab results: scopes
  • lab testedReverse-order cleanup confirmed: three tasks inserted A→B→C ran C→B→A. innerScope contents died with the parent; deriveScope contents survived parent cleanup and required their own doCleanup, exactly as documented.
  • Destruction is silent. Calling :set() or peek() on a destroyed Value raised no error: both no-op quietly. "Irrecoverable" does not mean "loud": Fusion will not tell you you're using a dead object. The only loud failure is 0.2-style :get(), which throws [Fusion] `StateObject:get()` has been replaced by `use()` and `peek()`. Treat use-after-destroy as a bug class your framework's structure must prevent, because the runtime won't catch it.
  • The possiblyOutlives warning did not fire for an outer-scope computed reading an inner-scope value in our bench; don't rely on the lifetime analysis catching every inversion.
Framework rule 1. One root scope per app, created at bootstrap. One innerScope per screen or feature. Components receive their scope as a parameter and never create root scopes. Because destruction is irrecoverable, pooling or reusing destroyed screens is off the table; model show/hide separately from create/destroy (see §11).

3 · State primitives

verified 3–00.3 removed StateObject:get() entirely. Reads happen two ways, and the distinction is the heart of the reactive graph:

  • use(state): the function passed into Computed and For-object callbacks. Reads the value and registers a dependency, so the computed re-runs when it changes.
  • peek(state): reads without creating a dependency. For event handlers, one-shot logic, and anywhere outside reactive callbacks.
local coins = scope:Value(120)
local price = scope:Value(50)

local canAfford = scope:Computed(function(use)
    return use(coins) >= use(price) -- reactive reads
end)

local function onBuyClicked()
    if peek(canAfford) then -- non-reactive read
        Network.Buy:Fire(peek(price))
    end
end

verified 3–0Observers are for side effects (sounds, analytics, tweening non-UI systems). In 0.3 they're scope-owned (scope:Observer(state)), and their :onChange() connections auto-disconnect when the scope cleans up. Manual disconnection is only for early teardown. This means side-effect wiring inherits component and screen lifetimes for free, no leak bookkeeping.

Derived-state cookbook

Three small patterns that cover most day-to-day derivations:

-- 1. Boolean gates: derive capability, bind it everywhere
local canAfford = scope:Computed(function(use)
    return use(coins) >= use(price)
end)

-- 2. Formatted labels: keep formatting in one computed, not in bindings
local coinLabel = scope:Computed(function(use)
    return `⛁ {use(coins)}`
end)

-- 3. Cross-store joins: computeds can read any number of stores
local canEquip = scope:Computed(function(use)
    return use(player.Level) >= item.minLevel
        and table.find(use(player.OwnedIds), item.id) ~= nil
end)
Lab results: the graph is lazy and dependencies are dynamic
  • lab testedComputeds are lazily evaluated. A freshly constructed computed had run zero times; setting its dependency still left it at zero; the first peek() ran it once. Invalidation marks the graph dirty; computation happens on read (or when a binding reads it, see §7). Consequence: an unread computed costs nothing, and construction order doesn't trigger evaluation cascades.
  • Dependencies are per-evaluation. A state read with peek() inside a computed never invalidated it (changing that state caused zero recomputes). Dependency registration happens only through use(), only on the branches actually executed; this is what makes the gate pattern in §11 free.
  • Observer auto-disconnect confirmed: after doCleanup, further :set() calls fired nothing. Bonus: Observer:onBind(fn) is supported and fires once immediately plus on every change: the right tool for "sync this side effect now and henceforth".

Sources: computeds, observers, discussion #217

4 · Component conventions

verified 3–0Fusion has no class or component-object system. A component is a plain function with the signature (scope, props) returning UI content. Scope goes first deliberately, so user components compose exactly like built-ins:

local function Button(scope: Fusion.Scope<typeof(Fusion)>, props: {
    Text: Fusion.UsedAs<string>?,
    Disabled: Fusion.UsedAs<boolean>?,
    OnClick: (() -> ())?,
})
    return scope:New "TextButton" {
        Text = props.Text,
        AutoButtonColor = scope:Computed(function(use)
            return not use(props.Disabled)
        end),
        [OnEvent "Activated"] = function()
            if props.OnClick and not peek(props.Disabled) then
                props.OnClick()
            end
        end,
    }
end

-- usage, after adding Button to the scope's method tables:
local ui = scope:Button {
    Text = "Buy",
    Disabled = canAfford,
    OnClick = onBuyClicked,
}

verified 3–0Props conventions that real component libraries converge on (and the docs endorse):

  • Reactive props accept both shapes: type them UsedAs<T> so callers can pass a raw value or a state object interchangeably.
  • Static config stays plain: e.g. TweenSettings: TweenInfo?. Don't make everything reactive.
  • Events are callback props: expose OnClick/Activated functions in the public API, not raw [OnEvent] keys. Special keys are an implementation detail of the component.
  • Children via a typed prop: e.g. Content: UsedAs<{Instance}>?, forwarded to [Children] internally.
  • Passthrough: an [any]: any index signature forwarding unrecognized properties to the underlying instance, so consumers can set LayoutOrder, Size, etc. without the component enumerating them.

Sources: components best practices, fusion-components, PluginEssentials (both 0.2; patterns translated)

Lab results: components and bindings
  • lab testedThe exact Button above (adapted) ran as written: scoped(Fusion, { Button = Button }) made scope:Button{ Text = state } work, and the instance tracked the state.
  • Bound properties apply synchronously. labelText:set("world") changed btn.Text on the very same line: no deferral, no frame delay. A bound property read is what forces a lazy computed to evaluate, which is why binding a computed makes it effectively eager (measured in §11).
  • [Children], [OnEvent] behaved as documented, and doCleanup on the owning scope destroyed the created instances.

5 · Project organization

verified 3–0Mature Fusion codebases use one module per component with layered category folders. fusion-components splits src/components into base/ primitives (button, frame, text, textBox, imageLabel, sinkFrame), common/ composed pieces, and utility/, with src/theme separate and Flipbook stories colocated in-repo but excluded from the published package. PluginEssentials separates host-instance wrappers from visual components, with shared helpers isolated in Util/.

ReplicatedStorage/Client/UI/
├── components/
│   ├── base/        -- primitives: Button, Frame, Text, Image, Input
│   ├── common/      -- composed: ItemCard, CurrencyLabel, TabBar
│   └── utility/     -- non-visual helpers: Padding, Corner, Stroke
├── screens/         -- one module per screen, (scope, props) like any component
├── stores/          -- reactive state modules, constructed with the app scope
├── theme/           -- palette, type scale, spacing tokens
├── stories/         -- dev-only component stories, excluded from ship
└── App.luau         -- bootstrap: root scope, stores, router, ScreenGui
Framework rule 2. Primitives depend on theme only. Composed components depend on primitives. Screens depend on components and stores. Stores depend on nothing UI-side. Dependencies point one way, and the scope flows down through every layer as the first argument.

Two moderating notes so this layout doesn't get over-applied. First, communitynot everything is a component: inside a screen, name one-off sub-trees as local variables and local functions rather than promoting them to modules; component modules are for things that get reused or need stories (see the field notes for the practitioner thread behind this). Second, the theme/ layer here means design tokens, a frozen table of colors, fonts, and spacing constants, not a theming system; most Roblox games ship exactly one visual identity, so keep the constants and skip the switching machinery (see "On theming" in the field notes).

6 · Stores and server data

verified 3–0The official docs explicitly warn against module-level global state: "you can't destroy globals until the very end of your program," plus hidden dependencies and untestability. The sanctioned alternative for widely-shared values is the Contextual API: Fusion.Contextual(default), temporarily overridden with :is(value):during(callback) and read with :now(). The docs frame globals as risky-but-permitted, not forbidden.

Source: sharing values

synthesisThe pattern that follows from the verified primitives: store factories instead of store globals:

-- stores/PlayerStore.luau
local function PlayerStore(scope)
    local coins = scope:Value(0)
    local level = scope:Value(1)
    return {
        Coins = coins,
        Level = level,
        -- single-writer discipline: mutations only through methods
        ApplySnapshot = function(data)
            coins:set(data.Coins)
            level:set(data.Level)
        end,
    }
end
return PlayerStore

-- App.luau
local scope = scoped(Fusion, { Button = Button, ... })
local stores = {
    Player = PlayerStore(scope),
}
-- hand stores to screens via props (explicit) or Contextual (ambient)

synthesisBridging server data: keep the network layer entirely outside components. One thin module owns the remotes and writes into stores; components never see a RemoteEvent.

-- stores/replication.luau: the only file that touches remotes
Network.PlayerData.OnClientEvent:Connect(function(snapshot)
    table.freeze(snapshot)          -- see §10: frozen tables skip updates
    stores.Player.ApplySnapshot(snapshot)
end)

Data flows one direction: remote → store Value → Computed → bound property. UI actions go the other way as plain function calls that fire remotes. Nothing in the reactive graph ever writes to the server directly.

Lab results: Contextual
  • lab testedContextual behaves exactly as documented, including nesting: :now() returned the default outside, the override inside :is():during(), and correct inner/outer values in nested overrides.
  • The lazy-evaluation trap. A computed constructed during an override but first read after it ended returned the default, not the override, because computeds are lazy (§3), ctx:now() runs at evaluation time, and by then the :during() block is long gone. If you use Contextual for dependency injection (themes, stores), read it in construction-time code (the component function body) and capture the result, rather than calling :now() inside computed callbacks.

7 · Instances and bindings

official docsInstances are built with scope:New "ClassName" { ... }. String keys assign properties; if the assigned value is a state object, the property re-assigns on every change; that's the entire binding system, no VDOM or diffing. Special keys available in 0.3: [Children], [Child], [OnEvent], [OnChange], [Out], [Attribute], [AttributeChange], [AttributeOut]. Gone from 0.2: [Ref] (constructors return the instance; just assign it to a local) and [Cleanup] (scopes own lifetimes now).

local hpBar = scope:New "Frame" {
    Size = scope:Computed(function(use)
        return UDim2.fromScale(use(healthPercent), 1)
    end),
    [OnChange "AbsoluteSize"] = function(newSize)
        -- fires with the new value when the property changes
    end,
    [Children] = {
        scope:New "UICorner" { CornerRadius = UDim.new(0, 6) },
    },
}

official docs[Out "Property"] writes property changes back into a Value: instance-to-state, one direction. Fusion has no true two-way binding primitive; binding a property and outputting the same property means two systems fighting over one value. The clean pattern for inputs: bind nothing, read via [Out], or set state only in event handlers.

Sources: New, Out, OnEvent

Lab results: Hydrate, Out, and attributes
  • lab testedHydrate does not take ownership of the instance. It returned the same instance, applied and reactively updated bound properties; after doCleanup, the bindings died but the instance itself survived, fully usable. That makes Hydrate the right tool for wrapping pre-built instance trees (Studio-designed or generated UI) in reactive state without Fusion claiming their lifetimes.
  • [Out "Text"] captured the initial property value immediately and tracked later property writes. Binding and outputting the same property to the same Value was stable in both directions in our bench: state writes reached the property and property writes reached the state, with no feedback loop. But treat true two-way binding as a last resort; one-directional flow stays debuggable.
  • [Attribute] set and reactively updated instance attributes; [AttributeOut] reflected the current value. The attribute key family works as the API reference describes.

8 · List rendering

official docsNever rebuild children arrays inside a Computed; the For-objects exist precisely so list updates are incremental. All three take (scope, inputTable, processor) where inputTable can be a state object:

ConstructorProcessorUse when
ForValues(use, scope, value) → newValueTransforming values, keys preserved, e.g. item data → item card instance.
ForPairs(use, scope, key, value) → (newKey, newValue)Both sides matter, e.g. inventory keyed by slot.
ForKeys(use, scope, key) → newKeyOnly keys transform; docs note it reprocesses only affected keys.

official docsThe critical lifecycle detail: each processor invocation receives its own scope, cleaned up automatically when that item is reprocessed or removed. Per-item connections, springs, and sub-instances die with the item: no manual bookkeeping.

local items = scope:Value({ ... })  -- array of item data

local cards = scope:ForValues(items, function(use, itemScope, item)
    return itemScope:ItemCard { Item = item }  -- dies when item leaves
end)

scope:New "ScrollingFrame" { [Children] = cards }

Sources: ForValues, ForPairs, ForKeys

Lab results: incremental behavior confirmed, and the real scaling shape
  • lab testedRemoving one value from a 5-item ForValues ran the processor zero additional times and cleaned exactly the removed item's scope; adding one ran it exactly once. ForKeys over 100 keys reprocessed exactly 1 key when 1 was added. Per-item caching is real (frozen items are cached by identity; another reason to freeze). One caveat from a later run: on a removal, sibling items' scopes were sometimes recycled too, still without reprocessing; rely on processor-call counts for cost reasoning, and treat per-item cleanup timing as an implementation detail rather than a removal signal.
  • The per-update cost is the table diff, not the processing. At 1,000 items, a single-item add cost ~4.7 ms whether the processor returned plain data or built Frames: Fusion scans the new table against the old one, so updates scale O(n) with collection size even when one item changed. Cheap at game-UI scale, but don't put a 10,000-row table behind one For-object and update it per-frame.
  • ForPairs confirmed equally incremental: changing one value in a 3-pair dictionary reprocessed exactly that pair, and key/value inversion round-tripped correctly.
  • Whole-table churn is GC pressure, not a leak: replicating a community leak report (cycling a ForPairs input between 200 items and empty, 20 times), every destroyed frame was eventually collected with zero alive at rest, but up to 2,800 dead frames sat awaiting GC mid-churn with memory swinging tens of MB. Prefer incremental key changes over whole-table swaps at high frequency; details in the field notes.
  • Nested For-objects work, but budget for branch rebuilds. An outer ForValues of lists with an inner ForValues per list produced correct output and cascaded cleanup on removal. But removing one outer list also recycled the surviving list's per-item scopes in our bench (5 cleanups where the minimum would be 3). Don't assume nested structures rebuild only the removed branch; keep per-item work in nested Fors cheap or flatten the data.

9 · Animation

official docsBoth animation objects are state objects that follow a goal: you bind properties to them like any other state:

  • scope:Tween(goal, tweenInfo): follows the goal along an easing curve. tweenInfo itself accepts a state object, so easing can change dynamically.
  • scope:Spring(goal, speed, damping): physics-based following. speed and damping both accept state objects. Damping semantics: 0 = no resistance, 1 = critical (no overshoot), higher = overdamped. Doubling speed exactly halves motion time.
local menuOpen = scope:Value(false)

local slide = scope:Spring(scope:Computed(function(use)
    return if use(menuOpen) then UDim2.fromScale(0, 0) else UDim2.fromScale(-1, 0)
end), 25, 1)

scope:New "Frame" { Position = slide }  -- flip menuOpen; physics does the rest

The idiom: state describes where things should be; springs and tweens are interposed between logical state and bound properties. Animation becomes declarative: no imperative TweenService calls in components.

Lab results: animation
  • lab testedThe menu-slide pattern above ran verbatim: spring (speed 25, damping 1) reached 62% of its goal in 100 ms and settled by 500 ms; the bound property landed exactly on 1.0 / 0.0 after each flip.
  • Reactive parameters work mid-flight: boosting speed from 5 to 50 while the spring was at 14% snapped it to convergence. A reactive TweenInfo swap took effect on the next goal change. A linear 0.4 s tween measured 5.02/10 at its midpoint.
  • Springs keep the graph quiet. While the spring animated the bound property every frame, the goal computed evaluated only 3 times total; the per-frame motion lives inside the animation object, outside the dependency graph. Interpose springs/tweens freely; they don't multiply reactive work.
  • Interrupted tweens re-tween from their current value, a question the docs leave open, now measured: a linear 0→10 tween interrupted at 5.05 by a new goal of 0 descended smoothly (3.80 a tenth of a second later) and settled at 0. No snapping, no restart-from-goal. Retargeting mid-flight is safe.
  • Typed springs confirmed: UDim2 and Color3 goals both animated and converged; the "animatable types" phrase covers Roblox's composite value types in practice.
  • Settled springs cost nothing in our bench, despite community reports that they never sleep (issue #370): 1,000 property-bound settled springs added +0.02 ms per frame, and underdamped ones returned to baseline within 3 seconds. Animating springs are the real cost (+8.5 ms for 1,000 in motion). Details and the fork story in the field notes.

Sources: Tween, Spring

10 · Performance

official docsFusion skips updates that aren't "meaningfully different" via a similarity test. The rules, verbatim from the optimisation guide:

  • Different types never pass similarity.
  • Frozen tables and tables with metatables pass when == to each other.
  • Unfrozen, non-metatable tables never pass: they always propagate updates, even when contents are identical.
  • Everything else passes when == (NaN and userdata per == semantics).
Framework rule 3. "Freeze every table you create, unless you need to change what's inside later", the docs' words. Freezing is near-zero cost and lets Fusion skip entire downstream branches. Make table.freeze the default in stores, computed returns, and network payloads.

Beyond similarity: keep computeds granular (one value per concern, so a coin change doesn't recompute a whole HUD table), and use For-objects for collections (§8) so per-item work is incremental rather than whole-list.

Lab results: the similarity engine, measured
OperationDownstream recompute?
set() the same unfrozen table reference againYes: unfrozen tables never pass similarity, even by identity
set() the same frozen reference againNo, skipped
set() a new frozen table with equal contentsYes, similarity is ==, not deep-equal
set(5) when the value is already 5No, skipped
  • lab testedComputeds are propagation firewalls. With src → mid → leaf where mid returns a boolean, changing src in a way that left mid's result identical caused zero leaf recomputes; flipping mid's result recomputed the leaf once. The "granular computeds" advice is not style: each computed boundary is a measured cutoff point.
  • Practical corollary: a store that mutates and re-sets the same unfrozen table (table.insert(t, x); state:set(t)) triggers every downstream update every time. Build a new frozen table per change instead.
Lab results: stress and error resilience
  • lab testedGraph scale is a non-issue at game-UI size. A 100-deep computed chain cold-evaluated in 0.30 ms and propagated an update in 0.12 ms. Setting 1,000 individually-bound Values took 3.5 ms total. One :set() fanned out to 1,000 observers in 2.8 ms. Architect for clarity; the graph won't be your bottleneck.
  • Errors degrade gracefully. A computed whose callback threw did not break reads: peek() returned the last good value (with one Fusion-logged error), and the computed recovered on the next valid input. A ForValues processor throwing for one item still produced all other items. An observer handler throwing didn't stop other observers from firing. Fusion's blast radius for a bad callback is that callback. But note the flip side: stale values and silent no-ops (§2) mean errors hide well. Watch the output console during development; the runtime is designed to keep going, not to stop you.

Source: optimisation

11 · Screens and routing

synthesisNo official routing guidance exists; the research produced no verified community pattern either. But the verified lifecycle constraints (§2: destroyed = irrecoverable; innerScope for child lifetimes) leave exactly two coherent architectures:

ArchitectureMechanismTrade-off
Persistent screensBuild every screen once at boot; bind each screen's Visible to a computed over a CurrentScreen value.Instant switching, screens keep internal state; full memory footprint always resident; bound properties on hidden screens still update unless gated.
Scoped create/destroyOne innerScope per screen; navigation cleans the old scope and constructs the new screen fresh.Clean memory, guaranteed teardown of connections/springs; loses transient screen state; construction cost on every open; a destroyed screen can never be revived; always construct fresh.
local function Router(appScope, screens)
    local current = appScope:Value(nil)
    local screenScope = nil
    local function navigate(name, props)
        if screenScope then doCleanup(screenScope) end
        screenScope = appScope:innerScope()
        local ui = screens[name](screenScope, props or {})
        ui.Parent = playerGui.Main
        current:set(name)
    end
    return { Current = current, Navigate = navigate }
end

synthesisRecommended hybrid for a typical game with a HUD plus ~10 menu screens: the HUD and any always-relevant chrome are persistent; menu screens (shops, inventories, settings) use scoped create/destroy: they're heavy, rarely all open, and their state should come from stores anyway, not from surviving instances. If a screen must keep transient state across closes, that state belongs in a store, which makes the destroy/recreate cycle lossless.

One extra gate for persistent screens: wrap expensive bound computeds so hidden screens go quiet, e.g. computeds that early-return when use(isOpen) is false, since property bindings keep evaluating on invisible instances.

Lab results: the router numbers
  • lab testedHidden screens are not free. A frame with Visible = false and one bound computed evaluated on 100 of 100 store updates: bindings ignore visibility entirely. The theoretical concern is a measured fact.
  • The gate pattern reduces that to zero. A computed that early-returns while use(open) is false ran its expensive branch 0 times across 100 store updates while closed (and 11 times across 10 updates once opened). The mechanism: dependencies register per-evaluation, so while the gate is closed the store isn't a dependency at all; updates don't even invalidate the computed. 
    Size = scope:Computed(function(use)
    if not use(isOpen) then
        return UDim2.new() -- store never use()d: no dependency while closed
    end
    return expensiveLayout(use(store))

  • Scoped create/destroy is cheap and airtight. A screen with 20 bound children plus an observer and a spring tore down completely on doCleanup(screenScope): instances destroyed, observer silent afterward. 100 create/destroy cycles of an 11-instance screen took 14 ms total (~0.14 ms per cycle) and left zero instances behind. Combined with §8's 7.5 ms build for 1,000 frames, construction cost is no argument against the create/destroy architecture at game-UI scale.

13 · Context: React-lua and Vide

synthesisNo claims in this area survived to verification, so treat this as background judgment rather than researched fact. The architecture-relevant distinction: React-lua re-renders component functions and diffs a virtual tree, so its frameworks organize around render purity and memoization; Fusion (like Vide, which shares the fine-grained reactive model) runs component functions once and lets the dependency graph update individual properties directly. Consequences for everything above: components are construction-time code (no re-render loop to design for), state granularity is the performance lever (not memo boundaries), and lifetimes are explicit via scopes (not implied by unmounting). If a pattern you're importing comes from React-lua material, check whether it exists to solve re-rendering; if so, it likely has no Fusion equivalent and can be dropped.

14 · Component workbench: UI-Labs

official docsUI-Labs is the storybook plugin for Roblox: the missing piece of the code-first workflow, letting you build and iterate components in isolation without running the game. Three pillars, per its docs: a hot reloader ("see your changes in real-time"), controls (adjust component props from a panel instead of editing code), and a sandboxed environment (stories run "virtually in an isolated environment, no conflicts or side effects"). It supports Fusion, React/Roact, Vide, and Iris story formats, plus generic function stories.

Installation and layout

Two parts: the Studio plugin (Creator Store asset 14293316215 or GitHub releases), and an optional utility package (Wally pepeeltoro41/ui-labs, npm @rbxts/ui-labs) that adds typed helpers and advanced controls; the docs note the plugin works "with no extra code" but the package "unlocks its full potential". The plugin presents a dock widget: a story explorer listing every .story module it finds in the place, a live viewport canvas where the selected story mounts (multi-mounting supported), and a controls panel for the story's declared props. Stories are ModuleScripts named with the .story suffix (e.g. Button.story), which is exactly what the stories/ folder in §5's layout holds, or colocate them next to components and exclude them from the shipped build.

The Fusion story format

official docsA Fusion story module returns a table: fusion (your Fusion instance), story (a function called once to mount), and optional controls. The story function receives props with three fields: props.target (the Frame to parent into), props.controls (each declared control arrives as a ready-made Fusion.Value, reactive out of the box), and props.scope (Fusion 0.3+). The scope integration is the important part: "this scope will be destroyed with Fusion.doCleanup when the story is unmounted". UI-Labs speaks native 0.3 lifecycle, so a story needs no manual cleanup. (On 0.2, returning a cleanup function is required.) Controls are declared as key → default value pairs; primitive types (string/number/boolean) are inferred, and the utility package adds advanced controls like dropdowns, sliders, and color pickers.

-- components/base/Button.story (ModuleScript)
local Fusion = require(game.ReplicatedStorage.Packages.Fusion)
local Button = require(script.Parent.Button)

return {
    fusion = Fusion,
    controls = {
        Text = "Buy",
        Disabled = false,
    },
    story = function(props)
        local scope = props.scope
        local btn = scope:Button {
            Text = props.controls.Text,         -- already a Fusion.Value
            Disabled = props.controls.Disabled, -- flip it in the panel, UI reacts
        }
        btn.Parent = props.target
    end,
}

Note what this rewards: because §4's props convention types every reactive prop as UsedAs<T>, the control Values pass straight into your components with zero adaptation. The generic fallback format (for non-Fusion helpers or quick fixtures) is even simpler: the module returns function(target: Frame) → cleanup?, with one documented sharp edge: if the mount function errors before returning its cleanup, leaked instances can require a Studio restart.

Lab results: the story contract, simulated
  • lab testedWe executed the exact story above under a simulated UI-Labs runner (scope created via scoped(Fusion), controls wrapped as scope:Values, story(props) called once, doCleanup on unmount): the component mounted into the target, panel-style control writes drove the UI reactively (Text and a computed over Disabled both updated), and doCleanup alone fully unmounted everything: no cleanup function needed, matching the documented 0.3 behavior.
  • Framework takeaway: treat every component as story-first. If a component can't be mounted with nothing but a scope, a target, and a props table of state objects, it has hidden dependencies; the story format is a free architecture test.
  • communityThe failure mode is real: a Roblox OSS member built components inside stories referencing props.controls directly, then couldn't extract them for the game. The channel's rule matches this section: components take plain props, only the story touches controls, and the component module comes first with the story as a preview.

PlayerStore: factory, single writer, frozen snapshots

local function PlayerStore(scope: Fusion.Scope<any>)
    local coins = scope:Value(0)
    local level = scope:Value(1)
    local ownedIds = scope:Value(table.freeze({}))

    local store = { Coins = coins, Level = level, OwnedIds = ownedIds }

    function store.ApplySnapshot(data)
        coins:set(data.Coins)
        level:set(data.Level)
        ownedIds:set(table.freeze(table.clone(data.OwnedIds)))
    end

    function store.Spend(amount): boolean
        local current = peek(coins)
        if current < amount then
            return false
        end
        coins:set(current - amount)
        return true
    end

    function store.AddOwned(id)
        local next = table.clone(peek(ownedIds))
        table.insert(next, id)
        ownedIds:set(table.freeze(next))
    end

    return store
end

Note what the store does not have: no computed factories, no getters, no events. State objects are exposed directly for reading; methods are the only writers. When a screen needs derived state (like "does the player own item 4"), it builds that computed in its own scope, so the derivation dies with the screen instead of accumulating on the app-lifetime store scope.

Field notes: how the Fusion community actually works

communitySourced from 394 messages in the Roblox OSS Discord's Fusion channel (April 22 to July 5, 2026). This is the ground truth this guide would otherwise assume: what practitioners do, what trips them up, and where the architecture above should stay humble. Testable claims were checked in the lab; the results are inline.

Maintenance mode and the fork ecosystem

Beyond the author's June statement (§1), the channel's own read from May: "Prepare to wait for a while [for 0.4], if it will even come. Fusion is in a pretty good state right now." Meanwhile a small fork ecosystem carries fixes upstream won't ship: plainsour/Fusion (spring sleeping re-enabled, detach fixes) and the actively developed plainsour/lemon Wally package (improved Contextual/Value types, scope:detach accepting tables, doCleanup calling disconnect/Disconnect on custom objects). Treat forks as options to evaluate against measured problems, not upgrades to install by default.

For-objects under churn

communityAn April report with a minimal repro: cycling a ForPairs input between 200 items and empty made "GUI memory climb and never drop," with the channel unable to explain it ("might be a bug"). A separate May report: v0.3's disassembly "destroys perfectly fine pairs which are then created again" without their dependency changing, and pairs are not automatically reused; one member wrote a custom reuse hook. lab testedWe ran the leak repro on stock v0.3-beta with a weak-table liveness tracker, 20 add-200/clear cycles: no permanent leak (0 frames alive at rest, all collected without cleanup), but the transient pressure is real: mid-churn, up to 2,800 destroyed frames sat awaiting garbage collection with memory swinging tens of MB, which is exactly what a Studio memory graph shows as "climbing." The destroy-and-recreate report matches our own bench, which twice observed sibling scopes recycled on removals (§8). The practical rules: prefer incremental key changes over whole-table swaps when updates are frequent, keep For processors cheap because they may re-run more than the minimum, and if the memory graph climbs during churn, check whether it settles at idle before calling it a leak.

Not everything is a component

The most instructive thread: a developer with "12 levels of indentation" and "500 components" asking how to stay sane. The channel's answers, in order of preference: assign sub-trees to local variables (constructors return instances, so local header = scope:New "Frame" {...} then compose them), pull repeated chunks into local functions in the same file, use do/end blocks for editor folding, and only promote something to a component module when it gets reused. As one member put it about their deep one-off tree: "but its not gonna get reused." One member also cautioned that moving helpers to other modules or wrapping them in OOP carries a performance cost in Luau; whether or not that holds in your case, the instinct is right. Component modules are for reuse and stories; locals are for readability. A 500-component file is a readability failure, not a reason to build more framework.

Scope discipline and per-screen state files

Asked whether a component should create its own scope or receive one, the channel's answer is unanimous and matches Framework rule 1: components are always passed a scope; they never make their own. What does own a scope, in community practice, is what one member calls a "state file": a per-screen backend module (their example: InventoryState holding the selected item, max capacity, and the items to display) that the screen's components bind to. That sits neatly between this guide's global stores (§6) and per-component state: game-wide data lives in stores, screen-lifetime data lives in a state file owned by the screen's scope, and micro-state (hover, focus) lives in the component.

State granularity and nested tables

Two independent threads converge on the same rule: one Value per independently-changing property beats one big table. A member trying to observe a nested key inside a table Value found there is no built-in way to avoid firing on sibling changes (the whole table is one similarity unit, §10); the channel's answers were "make the nested table its own Value" or write a custom path-derivation hook. Another member's advice for dynamic data like position, health, and cooldowns: separate Values per property. And on store shape, the pattern one member described for module-level state matches §6's single-writer factories: "I don't let other code access my state directly but rather they use wrapper functions to modify the state... it's safer... easier to read and distinguish what I'm doing."

Derive, don't observe

A beginner bound a property by watching a value with an Observer and manually setting the property. The correction given in-channel is the rule this guide's §3 implies, stated plainly: if a property can be expressed as a function of state, use a Computed binding; Observers are only for side effects that leave the reactive world (sounds, analytics, camera shakes). An Observer-plus-manual-set is always a smell when the target is an instance property.

The instant-reset tween idiom

Asked how to snap a tween back to its start without animating, three separate members converged on the same pattern this guide's lab confirmed reactive TweenInfo makes possible: hold the TweenInfo in state, and set a zero-duration info for the frame where you want a snap.

local info = scope:Value(TweenInfo.new(0.3))
local progress = scope:Tween(target, info)

local function snapTo(value)
    info:set(TweenInfo.new(0))  -- zero-duration: next change is instant
    target:set(value)
    task.defer(function()
        info:set(TweenInfo.new(0.3))
    end)
end

Springs and the sleep question

communityMembers report that "springs don't sleep" (issue #370), meaning settled springs keep consuming frame time, and recommend plainsour's fork (based on 0.3) which re-enables spring sleeping plus detach and cleanup fixes. lab testedThis did not reproduce on stock v0.3-beta in our bench: 2,000 unbound settled springs cost +0.02 ms per frame, 1,000 property-bound settled springs +0.02 ms, and even 1,000 underdamped springs (damping 0.2) returned to baseline within 3 seconds of settling; only actively animating springs cost real time (+8.5 ms for 1,000 in motion). The honest conclusion: measure in your own game before adopting a fork. If your profiler shows settled springs burning frames, the fork exists; do not switch on hearsay.

Sizing and scaling

A recurring question with no built-in answer: how do you scale offset-based UI across resolutions when everything is code? The community's tool of choice is usePx from pretty-fusion-utils, a scale-factor helper in the spirit of the classic px pattern. Its ergonomics have sharp edges (the px() call cannot be used inside a computed because of scope capture, so values get declared outside), and the in-channel recommendation for the cleanest high-performance shape is to surface the scale as one state and multiply inside your computeds:

local pxScale = px:getScale()

Size = scope:Computed(function(use)
    local scale = use(pxScale)
    return UDim2.fromOffset(DOT_SIZE.X * scale, DOT_SIZE.Y * scale)
end)

The alternative is staying Scale-based (UDim2.fromScale plus aspect-ratio constraints), which needs no helper at all. Pick one per project; the template uses offsets for legibility and is small enough to convert either way.

Code, templates, and the team question

"Is it better to have a static UI template and hydrate it, or do everything by code?" The channel's answer: either works, stay consistent. The regulars prefer full code (easier to reference and manipulate what you built), and this guide is written for that mode. But one member's caveat is worth quoting for any team: "I prefer doing it by code but as soon as you're working with non-programmers that idea gets thrown out the window," against which another argued for making non-programmers adapt ("workflows that are programmer oriented should be prioritized"). If your team lands on designer-built trees, Hydrate is the bridge (§7: bindings die on cleanup, instances survive), and the paid Codify plugin converts Studio-built UI into Fusion code.

Transient elements without ceremony

For notifications and other timed elements, the community pattern is exactly as boring as it should be: a component module, created into a container per event, torn down after a delay. When a member wondered about caching instead of destroy-and-recreate, the in-channel reply was the right default posture: "Tbh idk how much more performance that would bring. Do you have issues with frame drops or smth?" Given §11's measured 0.14 ms create/destroy cycles, build the simple version first.

UI-Labs discipline, confirmed the hard way

A newcomer built components inside stories referencing props.controls directly, then struggled to "transfer UI-Labs UI to Fusion." The channel's correction matches §14 exactly: components never see UI-Labs. Write the component module first, taking plain props; the story exists only to preview it, mapping controls to props at the boundary: "generally you would just start with ur component module from the beginning, and only have the story for previewing it." If removing UI-Labs from your place would break a component, the dependency points the wrong way.

Smaller habits worth adopting

  • Timers: Fusion has no clock. Drive a Value from an external loop (while task.wait(1) do timeLeft:set(...) end) and derive everything else from it. The channel's answer to "is there some way in fusion" was simply "yes, an external loop."
  • Dynamic data granularity: for frequently-changing data (position, health, cooldowns), keep a separate Value per property rather than one table state; this matches §10's granular-computed guidance from the similarity angle.
  • Player state lives in its own module, away from UI code; members with custom state managers bind UI to them exactly like §6's store bridge.
  • Fusion without its animations is still Fusion: asked whether Fusion is overkill alongside an existing animation library, the channel's answer: the point is reactivity and declarative instance trees; Tween and Spring are extras. Several members run their own animation hooks on top.
  • Gotchas that cost people an afternoon: FontFace takes a Font, not an Enum.Font (use Font.fromEnum(Enum.Font.X), as the template does); the InfiniteLoop error can be triggered by yielding code around a :set(), and one of its variants is not documented; the attribute special keys are easy to miss entirely (one member: "I've never seen it in the tutorials. I just looked for Attribute in the documentation"); and any example using :get() means the whole snippet predates 0.3 (§1).
  • Tooling edges: roblox-ts users have @rbxts/fusion (no JSX) and @rbxts/hotfusion (JSX support); property autocomplete tooling for 0.3 is spotty; Studio's new CSS-like styling system's relationship to Fusion props was asked and never answered (now in §16's open questions).

On theming

Fourteen days of practitioner conversation and not one mention of themes, which matches how Roblox games ship: one deliberate visual identity, no runtime theme switching. This guide's earlier sections keep a theme module because centralized design tokens (colors, fonts, spacing in one frozen table) cost nothing and keep 50 components consistent. That is not a theming system, and you should not build one: no theme objects threaded through props, no reactive palette state, no light/dark infrastructure, unless your game has an actual feature that needs it. If that day comes, the pieces already exist (a Contextual holding a token table, read at construction time per §6's trap warning); until then, constants are the correct amount of engineering.
